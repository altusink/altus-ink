## 🎯 PROJECT OVERVIEW
Build ALTUSINK.IO, a **premium private SaaS** for Altus Ink agency to manage tattoo artists and their bookings.
### Business Model
- **Private platform** — Invite-only for Altus Ink managed artists
- **30% agency fee** on all deposits (artist receives 70%)
- **Non-refundable deposits** — Never show full session price
- **90-day retention** before payout to artists
### Core Features
1. Multi-tenant architecture (subdomain per artist: `artist.altusink.io`)
2. 4 user roles: CEO (agency), Artist, Vendor 1/2, Vendor Coordinator
3. Real-time booking with 10-minute slot locking
4. Tour Mode (address hidden until payment)
5. Stripe payment integration
6. Email + WhatsApp notifications
7. Multi-language support (EN, PT-BR, PT-PT, ES, FR, DE, IT)
8. GDPR compliant with audit logging

## 🗄️ SUPABASE SCHEMA
Run this SQL in Supabase SQL Editor:
```sql
-- Enable extensions
CREATE EXTENSION IF NOT EXISTS "uuid-ossp";
-- ============================================
-- PROFILES (extends auth.users)
-- ============================================
CREATE TABLE public.profiles (
  id UUID PRIMARY KEY REFERENCES auth.users(id) ON DELETE CASCADE,
  email TEXT NOT NULL,
  full_name TEXT,
  phone TEXT,
  role TEXT NOT NULL DEFAULT 'artist' CHECK (role IN ('ceo', 'artist', 'vendor1', 'vendor2', 'coordinator')),
  avatar_url TEXT,
  is_active BOOLEAN DEFAULT true,
  created_at TIMESTAMPTZ DEFAULT NOW(),
  updated_at TIMESTAMPTZ DEFAULT NOW()
);
-- ============================================
-- ARTISTS (tenants)
-- ============================================
CREATE TABLE public.artists (
  id UUID PRIMARY KEY DEFAULT uuid_generate_v4(),
  user_id UUID REFERENCES public.profiles(id) ON DELETE CASCADE,
  subdomain TEXT UNIQUE NOT NULL,
  display_name TEXT NOT NULL,
  bio TEXT,
  instagram TEXT,
  deposit_amount DECIMAL(10,2) NOT NULL DEFAULT 100.00,
  currency TEXT DEFAULT 'EUR' CHECK (currency IN ('EUR', 'CHF', 'USD', 'BRL')),
  timezone TEXT DEFAULT 'Europe/Berlin',
  tour_mode_enabled BOOLEAN DEFAULT false,
  is_active BOOLEAN DEFAULT true,
  is_approved BOOLEAN DEFAULT false,
  stripe_account_id TEXT,
  theme_color TEXT DEFAULT '#F59E0B',
  font_preset TEXT DEFAULT 'Inter',
  created_at TIMESTAMPTZ DEFAULT NOW(),
  updated_at TIMESTAMPTZ DEFAULT NOW()
);
-- ============================================
-- ARTIST AVAILABILITY
-- ============================================
CREATE TABLE public.artist_availability (
  id UUID PRIMARY KEY DEFAULT uuid_generate_v4(),
  artist_id UUID REFERENCES public.artists(id) ON DELETE CASCADE,
  day_of_week INTEGER NOT NULL CHECK (day_of_week BETWEEN 0 AND 6),
  start_time TIME NOT NULL,
  end_time TIME NOT NULL,
  slot_duration_minutes INTEGER DEFAULT 60,
  is_active BOOLEAN DEFAULT true,
  created_at TIMESTAMPTZ DEFAULT NOW()
);
-- ============================================
-- CITY SCHEDULE (Tour Mode)
-- ============================================
CREATE TABLE public.city_schedule (
  id UUID PRIMARY KEY DEFAULT uuid_generate_v4(),
  artist_id UUID REFERENCES public.artists(id) ON DELETE CASCADE,
  city TEXT NOT NULL,
  country TEXT NOT NULL,
  country_code TEXT,
  full_address TEXT NOT NULL,
  venue_name TEXT,
  start_date DATE NOT NULL,
  end_date DATE NOT NULL,
  is_active BOOLEAN DEFAULT true,
  created_at TIMESTAMPTZ DEFAULT NOW()
);
-- ============================================
-- BOOKING LOCKS (10-minute hold)
-- ============================================
CREATE TABLE public.booking_locks (
  id UUID PRIMARY KEY DEFAULT uuid_generate_v4(),
  artist_id UUID REFERENCES public.artists(id) ON DELETE CASCADE,
  slot_datetime TIMESTAMPTZ NOT NULL,
  duration_minutes INTEGER DEFAULT 60,
  customer_email TEXT NOT NULL,
  customer_name TEXT NOT NULL,
  customer_phone TEXT,
  customer_instagram TEXT,
  payment_intent_id TEXT,
  status TEXT DEFAULT 'pending' CHECK (status IN ('pending', 'paid', 'expired', 'cancelled')),
  expires_at TIMESTAMPTZ NOT NULL,
  created_at TIMESTAMPTZ DEFAULT NOW()
);
-- ============================================
-- BOOKINGS
-- ============================================
CREATE TABLE public.bookings (
  id UUID PRIMARY KEY DEFAULT uuid_generate_v4(),
  artist_id UUID REFERENCES public.artists(id) ON DELETE CASCADE,
  lock_id UUID REFERENCES public.booking_locks(id),
  city_schedule_id UUID REFERENCES public.city_schedule(id),
  customer_email TEXT NOT NULL,
  customer_name TEXT NOT NULL,
  customer_phone TEXT,
  customer_instagram TEXT,
  slot_datetime TIMESTAMPTZ NOT NULL,
  duration_minutes INTEGER DEFAULT 60,
  deposit_amount DECIMAL(10,2) NOT NULL,
  currency TEXT DEFAULT 'EUR',
  status TEXT DEFAULT 'confirmed' CHECK (status IN ('confirmed', 'completed', 'cancelled', 'no_show')),
  notes TEXT,
  created_at TIMESTAMPTZ DEFAULT NOW(),
  updated_at TIMESTAMPTZ DEFAULT NOW()
);
-- ============================================
-- PAYMENTS
-- ============================================
CREATE TABLE public.payments (
  id UUID PRIMARY KEY DEFAULT uuid_generate_v4(),
  artist_id UUID REFERENCES public.artists(id),
  booking_id UUID REFERENCES public.bookings(id),
  lock_id UUID REFERENCES public.booking_locks(id),
  amount DECIMAL(10,2) NOT NULL,
  currency TEXT DEFAULT 'EUR',
  payment_method TEXT DEFAULT 'stripe',
  payment_intent_id TEXT UNIQUE,
  status TEXT DEFAULT 'pending' CHECK (status IN ('pending', 'succeeded', 'failed', 'refunded')),
  error_message TEXT,
  created_at TIMESTAMPTZ DEFAULT NOW()
);
-- ============================================
-- DEPOSITS (with 90-day retention)
-- ============================================
CREATE TABLE public.deposits (
  id UUID PRIMARY KEY DEFAULT uuid_generate_v4(),
  payment_id UUID REFERENCES public.payments(id),
  artist_id UUID REFERENCES public.artists(id),
  booking_id UUID REFERENCES public.bookings(id),
  amount DECIMAL(10,2) NOT NULL,
  currency TEXT DEFAULT 'EUR',
  platform_fee DECIMAL(10,2) NOT NULL, -- 30%
  artist_amount DECIMAL(10,2) GENERATED ALWAYS AS (amount - platform_fee) STORED,
  is_refundable BOOLEAN DEFAULT false, -- Always false
  status TEXT DEFAULT 'held' CHECK (status IN ('held', 'available', 'released', 'forfeited')),
  retention_until TIMESTAMPTZ,
  released_at TIMESTAMPTZ,
  created_at TIMESTAMPTZ DEFAULT NOW()
);
-- Trigger: Set retention_until to 90 days
CREATE OR REPLACE FUNCTION set_retention_date()
RETURNS TRIGGER AS $$
BEGIN
  NEW.retention_until := NOW() + INTERVAL '90 days';
  RETURN NEW;
END;
$$ LANGUAGE plpgsql;
CREATE TRIGGER deposit_retention_trigger
BEFORE INSERT ON public.deposits
FOR EACH ROW EXECUTE FUNCTION set_retention_date();
-- ============================================
-- NOTIFICATIONS
-- ============================================
CREATE TABLE public.notifications (
  id UUID PRIMARY KEY DEFAULT uuid_generate_v4(),
  user_id UUID REFERENCES public.profiles(id),
  artist_id UUID REFERENCES public.artists(id),
  booking_id UUID REFERENCES public.bookings(id),
  type TEXT NOT NULL CHECK (type IN ('email', 'whatsapp', 'push', 'sms')),
  template TEXT NOT NULL,
  recipient TEXT NOT NULL,
  subject TEXT,
  data JSONB DEFAULT '{}',
  status TEXT DEFAULT 'pending' CHECK (status IN ('pending', 'sent', 'failed')),
  attempts INTEGER DEFAULT 0,
  sent_at TIMESTAMPTZ,
  scheduled_for TIMESTAMPTZ DEFAULT NOW(),
  error_message TEXT,
  created_at TIMESTAMPTZ DEFAULT NOW()
);
-- ============================================
-- FINANCIAL LEDGER
-- ============================================
CREATE TABLE public.financial_ledger (
  id UUID PRIMARY KEY DEFAULT uuid_generate_v4(),
  artist_id UUID REFERENCES public.artists(id),
  transaction_type TEXT NOT NULL,
  amount DECIMAL(10,2) NOT NULL,
  currency TEXT DEFAULT 'EUR',
  description TEXT,
  reference_type TEXT,
  reference_id UUID,
  created_at TIMESTAMPTZ DEFAULT NOW()
);
-- ============================================
-- AUDIT LOG
-- ============================================
CREATE TABLE public.audit_log (
  id UUID PRIMARY KEY DEFAULT uuid_generate_v4(),
  user_id UUID,
  artist_id UUID,
  action TEXT NOT NULL,
  table_name TEXT,
  record_id UUID,
  old_data JSONB,
  new_data JSONB,
  ip_address INET,
  user_agent TEXT,
  created_at TIMESTAMPTZ DEFAULT NOW()
);
-- ============================================
-- GLOBAL SETTINGS
-- ============================================
CREATE TABLE public.global_settings (
  id UUID PRIMARY KEY DEFAULT uuid_generate_v4(),
  key TEXT UNIQUE NOT NULL,
  value TEXT NOT NULL,
  description TEXT,
  is_public BOOLEAN DEFAULT false,
  updated_at TIMESTAMPTZ DEFAULT NOW()
);
-- Insert defaults
INSERT INTO public.global_settings (key, value, description, is_public) VALUES
  ('platform_fee_percentage', '30', 'Agency fee (Altus Ink)', false),
  ('retention_days', '90', 'Days before deposit release', false),
  ('lock_duration_minutes', '10', 'Booking lock duration', true);
-- ============================================
-- ROW LEVEL SECURITY
-- ============================================
ALTER TABLE public.profiles ENABLE ROW LEVEL SECURITY;
ALTER TABLE public.artists ENABLE ROW LEVEL SECURITY;
ALTER TABLE public.bookings ENABLE ROW LEVEL SECURITY;
ALTER TABLE public.booking_locks ENABLE ROW LEVEL SECURITY;
ALTER TABLE public.payments ENABLE ROW LEVEL SECURITY;
ALTER TABLE public.deposits ENABLE ROW LEVEL SECURITY;
-- CEO sees everything
CREATE POLICY "CEO full access" ON public.artists FOR ALL
  USING ((SELECT role FROM public.profiles WHERE id = auth.uid()) = 'ceo');
-- Artists see only their own
CREATE POLICY "Artists own data" ON public.artists FOR ALL
  USING (user_id = auth.uid());
-- Public booking pages (read-only, limited fields)
CREATE POLICY "Public artist profiles" ON public.artists FOR SELECT
  USING (is_active = true AND is_approved = true);
-- Indexes for performance
CREATE INDEX idx_bookings_artist_date ON public.bookings(artist_id, slot_datetime);
CREATE INDEX idx_locks_expires ON public.booking_locks(expires_at) WHERE status = 'pending';
CREATE INDEX idx_deposits_retention ON public.deposits(retention_until) WHERE status = 'held';
```
